const icons = dv.current().file.lists.map((l) => l.text);

await dv.table(
  ["Icon", "Name"],
  icons.map((l) => ["", l])
);

dv.container.querySelectorAll("tbody tr").forEach((tr, index) => {
  const td = tr.querySelector("td");
  obsidian.setIcon(td, icons[index], 16);
});
